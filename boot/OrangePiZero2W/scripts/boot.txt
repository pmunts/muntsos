mmc dev 0

# Flash boot loader to SPI flash, if u-boot-sunxi-with-spl.bin is present
# in the FAT partition.

if test -e mmc 0:1 u-boot-sunxi-with-spl.bin ; then
  echo ""
  echo "Writing u-boot-sunxi-with-spl.bin to SPI flash..."
  echo ""

  mmc dev 0
  load mmc 0:1 0x40300000 u-boot-sunxi-with-spl.bin
  sf probe
  sf erase 0 +0xf0000
  sf write 0x40300000 0 $filesize

  echo ""
  echo "Replace the reflash card with a MuntsOS boot card and then reboot."
  echo ""

  exit
fi

# Allocate some memory buffers

setexpr INBUF  $loadaddr + 0x000000
setexpr DTB    $loadaddr + 0x100000
setexpr KERNEL $loadaddr + 0x200000

# Import some target settings from config.txt

load mmc 0:1 $INBUF config.txt
env import -t -r $INBUF ${filesize}

# Load device tree

load mmc 0:1 $DTB $boardname.dtb
fdt addr $DTB
fdt resize 65536

# Load device tree overlays

for OVL in ${OVERLAYS} ; do
  load mmc 0:1 $INBUF overlays/${OVL}.dtbo
  fdt apply $INBUF
done

# Load kernel files

load mmc 0:1 $KERNEL $boardname.img

# Boot the Linux kernel

setenv bootargs $CMDLINE BOARDNAME=$boardname
booti $KERNEL - $DTB
